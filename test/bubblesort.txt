def bubble(list: TNumberList)
  len = list."长度"
  i = 0
  while i < len - 1 repeat
    j = 0
    while j < len - 1 - i repeat
      if list第j个元素 > list第j+1个元素 then
        temp = list第j+1个元素
        list第j+1个元素 = list第j个元素

  end
end
